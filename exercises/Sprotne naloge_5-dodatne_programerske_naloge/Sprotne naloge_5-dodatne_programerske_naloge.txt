1) Naloga

V pomnilniku sta definirani dve spremenljivki:

            .align
STEV1:	.word	0xFF112200
STEV2:	.byte	0xAA
            .align
REZ:     .space	4

Z uporabo baznega naslavljanja z odmikom, zapišite število STEV2 na dva najnižja bajta 32-bitnega števila STEV1. Nato vsebino spremenljivke STEV1 kopirajte na naslov spremenljivke REZ. Najnižja bajta števila STEV1 sta v zaporedju 0x00 in 0x22.
          .text
          .align
STEV1:	  .word	0xFF112200
STEV2:	  .byte	0xAA
          .align
REZ:      .space	4

          .global __start

__start:
          adr r0,STEV1
          adr r1,STEV2
          ldrb r3,[r1]
          strb r3,[r0]
          strb r3,[r0,#1]
          adr r1,REZ
          ldr r2,[r0]
          str r2,[r1]
          
          
       
    
          
    
          
__end:    b __end   













2) Naloga

V pomnilniku sta podani dve 32-bitni spremenljivki:

            .align
STEV1:	.word	0x66774455
STEV2:	.word	0x22330011

a) Zamenjaj spodnjih 16 bitov številke STEV1 z zgornjimi šestnajstimi biti številke STEV2.
b) Zamenjaj zgornjih 16 bitov številke STEV1 s spodnjimi šestanjstimi biti številke STEV2.

Uporabi zgolj bazno naslavljanje z odmikom.



          .text
          .align
STEV1:	  .word	0x66774455
STEV2:	  .word	0x22330011
          .global __start

__start:  
          @a
          adr r0,STEV1
          adr r1,STEV2
          ldrh r2,[r1,#2]
          strh r2,[r0]
          
          @b
          adr r0,STEV1
          adr r1,STEV2
          ldrh r2,[r1]
          strh r2,[r0,#2]
          
          
          
       
    
          
    
          
__end:    b __end
